#Rails Server Başlatma

cd ZehraClinic/backend
rails server -b 0.0.0.0 -p 3000

#Arkada Çalışan Portu Kapatma

kill -9 $(lsof -t -i:3000)
kill -9 $(lsof -t -i:4000)

#Mobili Başlatma

cd ZehraClinic/mobil-app
npx expo start
